126 bool Welder::computeAST(const std::string& filename, const std::string& code)
127 {
128 try
129 {
130 m_parser.process(filename, code);
131 m_computed_ast = m_parser.ast();
132
133 if ((m_features & FeatureImportSolver) != 0)
134 {
135 m_import_solver.setup(m_root_file, m_parser.imports());
136 m_import_solver.process(m_computed_ast);
137 m_computed_ast = m_import_solver.ast();
138 }
139
140 if ((m_features & FeatureMacroProcessor) != 0)
141 {
142 m_macro_processor.process(m_computed_ast);
143 m_computed_ast = m_macro_processor.ast();
144 }
145
146 if ((m_features & FeatureNameResolver) != 0)
147 {
148 m_name_resolver.process(m_computed_ast);
149 m_computed_ast = m_name_resolver.ast();
150 }
151
152 if ((m_features & FeatureASTOptimizer) != 0)
153 {
154 m_ast_optimizer.process(m_computed_ast);
155 m_computed_ast = m_ast_optimizer.ast();
156 }
157
158 return true;
159 }
160 catch (const CodeError& e)
