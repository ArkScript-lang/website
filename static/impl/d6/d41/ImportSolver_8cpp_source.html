27 void ImportSolver::process(const Node& origin_ast)
28 {
29 m_logger.traceStart("process");
30
31 while (!m_imports.empty())
32 {
33 ImportWithSource source = m_imports.top();
34 m_logger.debug("Importing {}", source.import.toPackageString());
35
36 // Remove the top element to process the other imports
37 // It needs to be removed first because we might be adding
38 // other imports later and don't want to pop THEM
39 m_imports.pop();
40 const auto package = source.import.toPackageString();
41
42 if (m_packages.contains(package))
43 {
44 // merge the definition, so that we can generate valid Full Qualified Names in the name & scope resolver
45 m_packages[package].import.with_prefix |= source.import.with_prefix;
46 m_packages[package].import.is_glob |= source.import.is_glob;
47 for (auto&& symbol : source.import.symbols)
48 m_packages[package].import.symbols.push_back(symbol);
49 }
50 else
51 {
52 // NOTE: since the "file" (=root) argument doesn't change between all calls, we could get rid of it
53 std::vector<ImportWithSource> temp = parseImport(source.file, source.import);
54 for (auto& additional_import : std::ranges::reverse_view(temp))
55 m_imports.push(additional_import);
56 }
57 }
58
59 m_logger.traceStart("findAndReplaceImports");
60 m_ast = findAndReplaceImports(origin_ast).first;
61 m_logger.traceEnd();
62
63 m_logger.traceEnd();
64 }

66 std::pair<Node, bool> ImportSolver::findAndReplaceImports(const Node& ast)
67 {
68 Node x = ast;
69 if (x.nodeType() == NodeType::List)
70 {
71 if (x.constList().size() >= 2 && x.constList()[0].nodeType() == NodeType::Keyword &&
72 x.constList()[0].keyword() == Keyword::Import)
73 {
74 // compute the package string: foo.bar.egg
75 const auto import_node = x.constList()[1].constList();
76 const std::string package = std::accumulate(
77 std::next(import_node.begin()),
78 import_node.end(),
79 import_node[0].string(),
80 [](const std::string& acc, const Node& elem) -> std::string {
81 return acc + "." + elem.string();
82 });
83
84 // if it wasn't imported already, register it
85 if (std::ranges::find(m_imported, package) == m_imported.end())
86 {
87 m_imported.push_back(package);
88 // modules are already handled, we can safely replace the node
89 x = m_packages[package].ast;
90 if (!m_packages[package].has_been_processed)
91 {
92 const auto import = m_packages[package].import;
93
94 // prefix to lowercase ; usually considered unsafe (https://devblogs.microsoft.com/oldnewthing/20241007-00/?p=110345)
95 // but we are dealing with prefix from filenames, thus we can somewhat assume we are in safe zone
96 std::string prefix = import.prefix;
97 std::ranges::transform(
98 prefix, prefix.begin(),
99 [](auto c) {
100 return std::tolower(c);
101 });
102
103 x = Node(Namespace {
104 .name = prefix,
105 .is_glob = import.is_glob,
106 .with_prefix = import.with_prefix,
107 .symbols = import.symbols,
108 .ast = std::make_shared<Node>(findAndReplaceImports(x).first) });
109
110 x.arkNamespace().ast->setPositionFrom(ast);
111 }
112 // we parsed an import node, return true in the pair to notify the caller
113 return std::make_pair(x, /* is_import= */ true);
114 }
115
116 // Replace by empty node to avoid breaking the code gen
117 x = Node(NodeType::List);
118 x.push_back(Node(Keyword::Begin));
119 }
120 else
121 {
122 for (std::size_t i = 0; i < x.constList().size(); ++i)
123 {
124 auto [node, is_import] = findAndReplaceImports(x.constList()[i]);
125 x.list()[i] = node;
126 }
127 }
128 }
129
130 return std::make_pair(x, /* is_import= */ false);
131 }

138 std::vector<ImportSolver::ImportWithSource> ImportSolver::parseImport(const std::filesystem::path& source, const Import& import)
139 {
140 m_logger.traceStart(fmt::format("parseImport {}", source.string()));
141
142 const auto path = findFile(source, import);
143 if (path.extension() == ".arkm") // Nothing to import in case of modules
144 {
145 // Creating an import node that will stay there when visiting the AST and
146 // replacing the imports with their parsed module
147 auto module_node = Node(NodeType::List);
148 module_node.push_back(Node(Keyword::Import));
149
150 auto package_node = Node(NodeType::List);
151 std::ranges::transform(
152 import.package,
153 std::back_inserter(package_node.list()), [](const std::string& stem) {
154 return Node(NodeType::String, stem);
155 });
156 module_node.push_back(package_node);
157 // empty symbols list
158 module_node.push_back(Node(NodeType::List));
159
160 m_packages[import.toPackageString()] = Package {
161 module_node,
162 import,
163 true
164 };
165
166 return {};
167 }
168
169 Parser parser(m_debug_level);
170 const std::string code = Utils::readFile(path.generic_string());
171 parser.process(path.string(), code);
172 m_packages[import.toPackageString()] = Package {
173 parser.ast(),
174 import,
175 false
176 };
177
178 m_logger.traceEnd();
179
180 auto imports = parser.imports();
181 std::vector<ImportWithSource> output;
182 std::ranges::transform(
183 imports,
184 std::back_inserter(output), [&path](const Import& i) {
185 return ImportWithSource { path, i };
186 });
187 return output;

199 std::filesystem::path ImportSolver::findFile(const std::filesystem::path& file, const Import& import_) const
200 {
201 const std::string package_path = import_.packageToPath();
202 if (auto maybe_path = testExtensions(m_root, package_path); maybe_path.has_value())
203 return maybe_path.value();
204
205 // search in all folders in environment path
